    <input type="password" id="password">
    <button onclick="login()">登入</button>
    <p id="error" style="color:red"></p>

  <div class="form-box hidden" id="formBox">
    <h2>請勾選上課日</h2>
    <form id="responseForm">
      <input type="hidden" id="studentName" name="姓名">
      <label><input type="checkbox" name="週二" value="✓"> 週二</label>
      <label><input type="checkbox" name="週四" value="✓"> 週四</label>
      <label><input type="checkbox" name="週五" value="✓"> 週五</label>
      <button type="submit">送出</button>

    <p id="success" style="color:green"></p>

  <script>
    const passwords = {
      "王邑東": "1234",
      "王邑西": "2234",
      "王邑南": "3234",
      "王邑北": "4234"
    };

    function login() {
      const name = document.getElementById("name").value;
      const pw = document.getElementById("password").value;
      const error = document.getElementById("error");

      if (passwords[name] === pw) {
        document.getElementById("loginBox").classList.add("hidden");
        document.getElementById("formBox").classList.remove("hidden");
        document.getElementById("studentName").value = name;
      } else {
        error.textContent = "姓名或密碼錯誤";
      }
    }

    // Google Apps Script 的表單提交網址（你會填入你自己的）
    const scriptURL = 'https://script.google.com/macros/s/AKfycby-7tYgeLfWVFYiApupNPHl9jZlwkJzqXqJk9HqJ-LW8b778IvcCWCU3T-8aj0eMe4X/exec';

    document.getElementById("responseForm").addEventListener("submit", e => {
      e.preventDefault();
      const form = e.target;
      fetch(scriptURL, { method: 'POST', body: new FormData(form) })
        .then(response => document.getElementById("success").textContent = "送出成功！")
        .catch(error => document.getElementById("success").textContent = "送出失敗！");
    });
  </script>
